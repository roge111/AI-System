–¢–∞–∫–∏—Ö –ø–∞—Ä —É –Ω–∞—Å –Ω–µ—Å–∫–æ–ª—å–∫–æ. –ò –Ω–∞ –∏—Ö –æ—Å–Ω–æ–≤–µ –º–µ—Ç–æ–¥ –ø—ã—Ç–∞–µ—Ç—Å—è –ø–æ—Å—Ç—Ä–æ–∏—Ç—å –Ω–µ–∫—É—é –ø—Ä—è–º—É—é, –ø—Ä–∏–±–ª–µ–∂–Ω–Ω—É—é –∫ –∏—Å—Ö–æ–¥–Ω—ã–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç–º.  –ï—Å–ª–∏ –≥–æ–≤–æ—Ä–∏—Ç—å –ø—Ä–æ—Å—Ç–æ - –Ω–∞–π—Ç–∏ –∑–∞–∫–æ–Ω–æ–º–µ—Ä–Ω–æ—Å—å –≤ –æ–±—É—á–∞—é—â–µ–π –≤—ã–±–æ—Ä–∫–µ.

–õ–∏–Ω–∏—è - —ç—Ç–æ –Ω–∞—à–∞ –ª–∏–Ω–µ–π–Ω–∞—è —Ä–µ–≥—Ä–µ—Å—Å–∏—è, –∞ —Ç–æ—á–∫–∏ - –≤—ã–±–æ—Ä–∫–∞. –ê –∫–∞–∫ —Ä–∞–±–æ—Ç–∞–µ—Ç –≤–Ω—É—Ç—Ä–∏? –î–∞–≤–∞–π—Ç–µ —Ä–∞–∑–±–µ—Ä–µ–º—Å—è.

–ú—ã —Ö–æ—Ç–∏–º, —á—Ç–æ–±—ã —Ä–∞–∑–Ω–∏—Ü–∞ –º–µ–∂–¥—É —Ä–µ–∞–ª—å–Ω—ã–º –∑–Ω–∞—á–µ–Ω–∏–µ–º Y –∏ –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–Ω—ã–º –±—ã–ª–∞ –∫–∞–∫ –º–æ–∂–Ω–æ –º–µ–Ω—å—à–µ. –¢–∞–∫–∞—è —Ä–∞–∑–Ω–∏—Ü–∞ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è `–æ—à–∏–±–∫–æ–π`. –î–ª—è —ç—Ç–æ–≥–æ –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –º–µ—Ç–æ–¥–∞ –Ω–∞–∏–º–µ–Ω—å—à–∏—Ö –∫–≤–∞–¥—Ä–∞—Ç–æ–≤ —Å —Ñ–æ—Ä–º—É–ª–æ–π:

$$ SSE = \left( \sum_{i=1}^n (Y_i - (a * X_i + b))^2 \right)$$

–ì–¥–µ:
- $Y_i$ - —Ä–µ–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ
- $a * X_i + b$ - –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ
-  n - –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ç–æ—á–µ–∫ –¥–∞–Ω–Ω—ã—Ö
-  a - –Ω–∞–∫–ª–æ–Ω
-  b - —Å–¥–≤–∏–≥

–ö–∞–∫ –Ω–∞–π—Ç–∏ a –∏ b ?
--------------------------------------------

a = $( n* ‚àë(X_i * Y_i) - ‚àë X_i * ‚àë Y_i)/(n * ‚àë (X_i) ^ 2 - ( ‚àëX_i)^2$

‚Äã
b = $(‚àëY_i - a * ‚àëX_i)/n$
‚Äã

# –ü—Ä–∏–º–∏—Ä–µ –Ω–∞ Python

```
import numpy as np

# –î–∞–Ω–Ω—ã–µ
X = np.array([50, 60, 70, 80, 90])
Y = np.array([200, 240, 280, 320, 360])

# –í—ã—á–∏—Å–ª—è–µ–º –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã
n = len(X)
a = (n * np.sum(X * Y) - np.sum(X) * np.sum(Y)) / (n * np.sum(X**2) - np.sum(X)**2)
b = (np.sum(Y) - a * np.sum(X)) / n

print(f"–ù–∞–∫–ª–æ–Ω (a): {a}")
print(f"–°–¥–≤–∏–≥ (b): {b}")

# –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–µ –¥–ª—è –ø–ª–æ—â–∞–¥–∏ 60
X_new = 60
Y_pred = a * X_new + b
print(f"–ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–Ω–∞—è —Ü–µ–Ω–∞ –¥–ª—è –ø–ª–æ—â–∞–¥–∏ {X_new} –º¬≤: {Y_pred}")
```

# –ú–µ—Ç–æ–¥ k-–±–ª–∏–∂–∞–π—à–∏—Ö —Å–æ—Å–µ–¥–µ–π

–ù—É –Ω–µ –≤—Å–µ –∂–µ —Ç–∞–∫ —Å—Ç—Ä–∞—à–Ω–æ. –ò–ª–∏ —Å—Ç—Ä–∞—à–Ω–æ? –ï—Å–ª–∏ —á—Ç–æ-—Ç–æ –Ω–µ –ø–æ–Ω—è—Ç–Ω–æ, —Ç–æ –Ω–∏–∂–µ –ø—Ä–∏–∫—Ä–µ–ø–ª—é –º–∞—Ç–µ—Ä–∏–∞–ª—ã. 

–ú–µ—Ç–æ–¥ k-–±–ª–∏–∂–∞–π—à–∏—Ö —Å–æ—Å–µ–¥–µ–π (k-NN) - —ç—Ç–æ –ø—Ä–æ—Å—Ç–µ–π—à–∏–π –º–µ—Ç–æ–¥ –∏ –∏–Ω—Ç—É–∏—Ç–∏–≤–Ω–æ –ø–æ–Ω—è—Ç–Ω—ã–π. –¢—É—Ç –Ω–µ —Ç–∞–∫ —Å—Ç—Ä–∞—à–Ω–æ. –õ–æ–≥–∏–∫–∞ —Ç–∞–∫–æ–≥–æ –º–µ—Ç–æ–¥–∞ –æ—á–µ–Ω—å –ø—Ä–æ—Å—Ç–∞. –ú —Å–º–æ—Ç—Ä–∏–º –Ω–∞ k –±–ª–∏–∂–∞–π—à–∏—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤ (—Å–æ—Å–µ–¥–µ–π) –∏ –Ω–∞ –∏—Ö –æ—Å–Ω–æ–≤–µ –¥–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç. –ú–µ—Ç–æ–¥ –ø—Ä–∏–Ω–∏–º –∫–∞–∫ –¥–ª—è –∑–∞–¥–∞—á –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏, —Ç–∞–∫ –∏ –¥–ª—è —Ä–µ–≥—Ä–µ—Å—Å–∏–∏.
–ï—Å–ª–∏ –≥–æ–≤–æ—Ä–∏—Ç—å –ø—Ä–æ –∑–∞–¥–∞—á—É –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏ - —Ç–æ –º—ã —Å–º–æ—Ç—Ä–∏–º –Ω–∞ k-–±–ª–∏–∂–∞–π—à–∏—Ö —Å–æ—Å–µ–¥–µ–π –∏ —Å–º–æ—Ç—Ä–∏–º, —á—Ç–æ —á–∞—â–µ –≤—Å—Ç—Ä–µ—á–∞–µ—Ç—Å—è. –ê –µ—Å–ª–∏ –≥–æ–≤–æ—Ä–∏—Ç—å –ø—Ä–æ —Ä–µ–≥–µ—Ä–µ—Å—Å–∏—é, —Ç–æ –º—ã –≤—ã—á–∏—Å–ª—è–µ–º –º–µ–¥–∏–∞–Ω—É.

–í–æ—Ç –∫–∞–∫ —ç—Ç–æ –º–æ–∂–Ω–æ —Ä–µ–∞–ª–∏–∑–æ–≤–∞—Ç—å –Ω–∞ `Python`

### –†–µ–∞–ª–∏–∑–∞—Ü–∏—è k-NN —Å –Ω—É–ª—è
–ú—ã —Å–æ–∑–¥–∞–¥–∏–º –∫–ª–∞—Å—Å `KNN`, –∫–æ—Ç–æ—Ä—ã–π –±—É–¥–µ—Ç:
1. –ü—Ä–∏–Ω–∏–º–∞—Ç—å –¥–∞–Ω–Ω—ã–µ –¥–ª—è –æ–±—É—á–µ–Ω–∏—è.
2. –í—ã—á–∏—Å–ª—è—Ç—å —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è –º–µ–∂–¥—É –æ–±—ä–µ–∫—Ç–∞–º–∏.
3. –ù–∞—Ö–æ–¥–∏—Ç—å `k` –±–ª–∏–∂–∞–π—à–∏—Ö —Å–æ—Å–µ–¥–µ–π.
4. –î–µ–ª–∞—Ç—å –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è (–¥–ª—è –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏ –∏ —Ä–µ–≥—Ä–µ—Å—Å–∏–∏).

### –ö–æ–¥ —Ä–µ–∞–ª–∏–∑–∞—Ü–∏–∏

```python
import numpy as np
from collections import Counter

class KNN:
    def __init__(self, k=3):
        self.k = k  # –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–æ—Å–µ–¥–µ–π

    def fit(self, X_train, y_train):
        """
        –û–±—É—á–µ–Ω–∏–µ –º–æ–¥–µ–ª–∏ (–ø—Ä–æ—Å—Ç–æ –∑–∞–ø–æ–º–∏–Ω–∞–µ–º –¥–∞–Ω–Ω—ã–µ).
        """
        self.X_train = X_train
        self.y_train = y_train

    def predict(self, X_test):
        """
        –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–µ –¥–ª—è –Ω–æ–≤—ã—Ö –¥–∞–Ω–Ω—ã—Ö.
        """
        predictions = [self._predict(x) for x in X_test]
        return np.array(predictions)

    def _predict(self, x):
        """
        –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–µ –¥–ª—è –æ–¥–Ω–æ–≥–æ –æ–±—ä–µ–∫—Ç–∞.
        """
        # 1. –í—ã—á–∏—Å–ª—è–µ–º —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è –¥–æ –≤—Å–µ—Ö –æ–±—ä–µ–∫—Ç–æ–≤ –≤ –æ–±—É—á–∞—é—â–µ–π –≤—ã–±–æ—Ä–∫–µ
        distances = [self._euclidean_distance(x, x_train) for x_train in self.X_train]

        # 2. –ù–∞—Ö–æ–¥–∏–º k –±–ª–∏–∂–∞–π—à–∏—Ö —Å–æ—Å–µ–¥–µ–π
        k_indices = np.argsort(distances)[:self.k]  # –ò–Ω–¥–µ–∫—Å—ã k –±–ª–∏–∂–∞–π—à–∏—Ö —Å–æ—Å–µ–¥–µ–π
        k_nearest_labels = [self.y_train[i] for i in k_indices]  # –ú–µ—Ç–∫–∏ k –±–ª–∏–∂–∞–π—à–∏—Ö —Å–æ—Å–µ–¥–µ–π

        # 3. –î–ª—è –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏: –≤–æ–∑–≤—Ä–∞—â–∞–µ–º –Ω–∞–∏–±–æ–ª–µ–µ —á–∞—Å—Ç–æ –≤—Å—Ç—Ä–µ—á–∞—é—â–∏–π—Å—è –∫–ª–∞—Å—Å
        most_common = Counter(k_nearest_labels).most_common(1)
        return most_common[0][0]

    def _euclidean_distance(self, x1, x2):
        """
        –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –µ–≤–∫–ª–∏–¥–æ–≤–∞ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è –º–µ–∂–¥—É –¥–≤—É–º—è —Ç–æ—á–∫–∞–º–∏.
        """
        return np.sqrt(np.sum((x1 - x2) ** 2))

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è
if __name__ == "__main__":
    # –î–∞–Ω–Ω—ã–µ –¥–ª—è –æ–±—É—á–µ–Ω–∏—è
    X_train = np.array([[1, 2], [2, 3], [3, 3], [6, 7], [7, 8], [8, 8]])
    y_train = np.array([0, 0, 0, 1, 1, 1])

    # –°–æ–∑–¥–∞—ë–º –º–æ–¥–µ–ª—å
    knn = KNN(k=3)
    knn.fit(X_train, y_train)

    # –ù–æ–≤—ã–π –æ–±—ä–µ–∫—Ç –¥–ª—è –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è
```

```python
import numpy as np

def entropy(y):
    """
    –í—ã—á–∏—Å–ª—è–µ—Ç —ç–Ω—Ç—Ä–æ–ø–∏—é –¥–ª—è –Ω–∞–±–æ—Ä–∞ –¥–∞–Ω–Ω—ã—Ö.
    """
    _, counts = np.unique(y, return_counts=True)  # –°—á–∏—Ç–∞–µ–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –æ–±—ä–µ–∫—Ç–æ–≤ –∫–∞–∂–¥–æ–≥–æ –∫–ª–∞—Å—Å–∞
    probabilities = counts / len(y)  # –í—ã—á–∏—Å–ª—è–µ–º –¥–æ–ª–∏
    return -np.sum(probabilities * np.log2(probabilities))  # –§–æ—Ä–º—É–ª–∞ —ç–Ω—Ç—Ä–æ–ø–∏–∏

# –í—ã—á–∏—Å–ª—è–µ–º —ç–Ω—Ç—Ä–æ–ø–∏—é
print(f"–≠–Ω—Ç—Ä–æ–ø–∏—è: {entropy(y):.3f}")
```

```
class DecisionTree:
    def __init__(self, max_depth=None):
        self.max_depth = max_depth  # –ú–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –≥–ª—É–±–∏–Ω–∞ –¥–µ—Ä–µ–≤–∞
        self.tree = None  # –°—Ç—Ä—É–∫—Ç—É—Ä–∞ –¥–µ—Ä–µ–≤–∞

    def fit(self, X, y, depth=0):
        """
        –û–±—É—á–µ–Ω–∏–µ –¥–µ—Ä–µ–≤–∞ —Ä–µ—à–µ–Ω–∏–π.
        """
        n_samples, n_features = X.shape
        n_classes = len(np.unique(y))

        # –£—Å–ª–æ–≤–∏—è –æ—Å—Ç–∞–Ω–æ–≤–∫–∏ —Ä–µ–∫—É—Ä—Å–∏–∏
        if (self.max_depth is not None and depth >= self.max_depth) or n_classes == 1:
            return np.argmax(np.bincount(y))  # –í–æ–∑–≤—Ä–∞—â–∞–µ–º —Å–∞–º—ã–π —á–∞—Å—Ç—ã–π –∫–ª–∞—Å—Å

        # –ò—â–µ–º –ª—É—á—à–∏–π –ø—Ä–∏–∑–Ω–∞–∫ –¥–ª—è —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è
        best_feature, best_threshold = self._best_split(X, y)

        if best_feature is None:
            return np.argmax(np.bincount(y))  # –ï—Å–ª–∏ —Ä–∞–∑–¥–µ–ª–∏—Ç—å –Ω–µ–ª—å–∑—è, –≤–æ–∑–≤—Ä–∞—â–∞–µ–º —Å–∞–º—ã–π —á–∞—Å—Ç—ã–π –∫–ª–∞—Å—Å

        # –†–∞–∑–¥–µ–ª—è–µ–º –¥–∞–Ω–Ω—ã–µ
        left_indices = X[:, best_feature] <= best_threshold
        right_indices = X[:, best_feature] > best_threshold

        # –†–µ–∫—É—Ä—Å–∏–≤–Ω–æ —Å—Ç—Ä–æ–∏–º –ª–µ–≤–æ–µ –∏ –ø—Ä–∞–≤–æ–µ –ø–æ–¥–¥–µ—Ä–µ–≤—å—è
        left_subtree = self.fit(X[left_indices], y[left_indices], depth + 1)
        right_subtree = self.fit(X[right_indices], y[right_indices], depth + 1)

        # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Å—Ç—Ä—É–∫—Ç—É—Ä—É –¥–µ—Ä–µ–≤–∞
        self.tree = {
            "feature": best_feature,
            "threshold": best_threshold,
            "left": left_subtree,
            "right": right_subtree
        }

        return self.tree

    def _best_split(self, X, y):
        """
        –ò—â–µ—Ç –ª—É—á—à–∏–π –ø—Ä–∏–∑–Ω–∞–∫ –∏ –ø–æ—Ä–æ–≥ –¥–ª—è —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è –¥–∞–Ω–Ω—ã—Ö.
        """
        n_samples, n_features = X.shape
        best_feature, best_threshold = None, None
        best_info_gain = -1

        # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –≤—Å–µ –ø—Ä–∏–∑–Ω–∞–∫–∏
        for feature in range(n_features):
            thresholds = np.unique(X[:, feature])  # –í—Å–µ —É–Ω–∏–∫–∞–ª—å–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è –ø—Ä–∏–∑–Ω–∞–∫–∞

            # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –≤—Å–µ –ø–æ—Ä–æ–≥–∏
            for threshold in thresholds:
                # –†–∞–∑–¥–µ–ª—è–µ–º –¥–∞–Ω–Ω—ã–µ
                left_indices = X[:, feature] <= threshold
                right_indices = X[:, feature] > threshold

                if len(left_indices) == 0 or len(right_indices) == 0:
                    continue  # –ü—Ä–æ–ø—É—Å–∫–∞–µ–º, –µ—Å–ª–∏ –æ–¥–Ω–∞ –∏–∑ –ø–æ–¥–≥—Ä—É–ø–ø –ø—É—Å—Ç–∞

                # –í—ã—á–∏—Å–ª—è–µ–º –ø—Ä–∏—Ä–æ—Å—Ç –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏
                info_gain = self._information_gain(y, y[left_indices], y[right_indices])

                # –û–±–Ω–æ–≤–ª—è–µ–º –ª—É—á—à–∏–π –ø—Ä–∏–∑–Ω–∞–∫ –∏ –ø–æ—Ä–æ–≥
                if info_gain > best_info_gain:
                    best_info_gain = info_gain
                    best_feature = feature
                    best_threshold = threshold

        return best_feature, best_threshold

    def _information_gain(self, y, y_left, y_right):
        """
        –í—ã—á–∏—Å–ª—è–µ—Ç –ø—Ä–∏—Ä–æ—Å—Ç –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏ –ø–æ—Å–ª–µ —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è.
        """
        # –≠–Ω—Ç—Ä–æ–ø–∏—è –¥–æ —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è
        entropy_parent = self._entropy(y)

        # –≠–Ω—Ç—Ä–æ–ø–∏—è –ø–æ—Å–ª–µ —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è
        n_left, n_right = len(y_left), len(y_right)
        n_total = n_left + n_right
        entropy_children = (n_left / n_total) * self._entropy(y_left) + (n_right / n_total) * self._entropy(y_right)

        # –ü—Ä–∏—Ä–æ—Å—Ç –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏–∏
        return entropy_parent - entropy_children

    def _entropy(self, y):
        """
        –í—ã—á–∏—Å–ª—è–µ—Ç —ç–Ω—Ç—Ä–æ–ø–∏—é –¥–ª—è –Ω–∞–±–æ—Ä–∞ –¥–∞–Ω–Ω—ã—Ö.
        """
        _, counts = np.unique(y, return_counts=True)
        probabilities = counts / len(y)
        return -np.sum(probabilities * np.log2(probabilities + 1e-10))  # –î–æ–±–∞–≤–ª—è–µ–º –º–∞–ª–µ–Ω—å–∫–æ–µ —á–∏—Å–ª–æ –¥–ª—è —Å—Ç–∞–±–∏–ª—å–Ω–æ—Å—Ç–∏

    def predict(self, X):
        """
        –ü—Ä–µ–¥—Å–∫–∞–∑—ã–≤–∞–µ—Ç –∫–ª–∞—Å—Å –¥–ª—è –Ω–æ–≤—ã—Ö –¥–∞–Ω–Ω—ã—Ö.

        return np.array([self._predict(x, self.tree) for x in X])

            return self._predict(x, tree["left"])  # –ò–¥—ë–º –≤ –ª–µ–≤–æ–µ –ø–æ–¥–¥–µ—Ä–µ–≤–æ
        else:
            return self._predict(x, tree["right"])  # –ò–¥—ë–º –≤ –ø—Ä–∞–≤–æ–µ –ø–æ–¥–¥–µ—Ä–µ–≤–æ

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è
if __name__ == "__main__":
```

2. **–°–∏–≥–º–æ–∏–¥–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è**:

### –û–±—É—á–µ–Ω–∏–µ –º–æ–¥–µ–ª–∏
–õ–æ–≥–∏—Å—Ç–∏—á–µ—Å–∫–∞—è —Ä–µ–≥—Ä–µ—Å—Å–∏—è –æ–±—É—á–∞–µ—Ç—Å—è –ø—É—Ç—ë–º –º–∏–Ω–∏–º–∏–∑–∞—Ü–∏–∏ **—Ñ—É–Ω–∫—Ü–∏–∏ –ø–æ—Ç–µ—Ä—å** (Log Loss). –§—É–Ω–∫—Ü–∏—è –ø–æ—Ç–µ—Ä—å –∏–∑–º–µ—Ä—è–µ—Ç, –Ω–∞—Å–∫–æ–ª—å–∫–æ –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–Ω—ã–µ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ –æ—Ç–ª–∏—á–∞—é—Ç—Å—è –æ—Ç —Ä–µ–∞–ª—å–Ω—ã—Ö –º–µ—Ç–æ–∫.

#### –§—É–Ω–∫—Ü–∏—è –ø–æ—Ç–µ—Ä—å (Log Loss)
$\[
\text{Log Loss} = -\frac{1}{N} \sum_{i=1}^{N} \left[ y_i \log(\sigma(z_i)) + (1 - y_i) \log(1 - \sigma(z_i)) \right]
\]$
–ì–¥–µ:
- $\( y_i \)$ ‚Äî —Ä–µ–∞–ª—å–Ω–∞—è –º–µ—Ç–∫–∞ (0 –∏–ª–∏ 1),
- $\( \sigma(z_i) \)$ ‚Äî –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–Ω–∞—è –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å,
- $\( N \)$ ‚Äî –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –æ–±—ä–µ–∫—Ç–æ–≤.

---

### –ö–æ–¥ –Ω–∞ Python

```

class LogisticRegression:
    def __init__(self, learning_rate=0.01, num_iterations=1000):
        self.learning_rate = learning_rate
        self.num_iterations = num_iterations
        self.weights = None
        self.bias = None
    
    def sigmoid(self, z):
        return 1 / (1 + np.exp(-z))
    
    def log_loss(self, y, y_pred):
        return -np.mean(y * np.log(y_pred) + (1 - y) * np.log(1 - y_pred))
    
    def fit(self, X, y):
        n_samples, n_features = X.shape
        self.weights = np.zeros(n_features)
        self.bias = 0
        
        for _ in range(self.num_iterations):
            linear_model = np.dot(X, self.weights) + self.bias
            y_pred = self.sigmoid(linear_model)
            
            dw = (1 / n_samples) * np.dot(X.T, (y_pred - y))
            db = (1 / n_samples) * np.sum(y_pred - y)
            
            self.weights -= self.learning_rate * dw
            self.bias -= self.learning_rate * db
    
    def predict(self, X):
        linear_model = np.dot(X, self.weights) + self.bias
        y_pred = self.sigmoid(linear_model)
        return [1 if i > 0.5 else 0 for i in y_pred]

```

–ö–æ–Ω–µ—á–Ω–æ! –î–∞–≤–∞–π —Ä–∞–∑–±–µ—Ä—ë–º —ç—Ç–æ—Ç –∫–æ–¥, –∫–æ—Ç–æ—Ä—ã–π —Ä–µ–∞–ª–∏–∑—É–µ—Ç **–ª–æ–≥–∏—Å—Ç–∏—á–µ—Å–∫—É—é —Ä–µ–≥—Ä–µ—Å—Å–∏—é —Å –Ω—É–ª—è**. –ú—ã –ø–æ—ç—Ç–∞–ø–Ω–æ –æ–±—ä—è—Å–Ω–∏–º, —á—Ç–æ –¥–µ–ª–∞–µ—Ç –∫–∞–∂–¥–∞—è —á–∞—Å—Ç—å –∫–æ–¥–∞.

### –ö–ª–∞—Å—Å `LogisticRegression`

#### 1. –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è (`__init__`)
```python
def __init__(self, learning_rate=0.01, num_iterations=1000):
    self.learning_rate = learning_rate
    self.num_iterations = num_iterations
    self.weights = None
    self.bias = None
```
- **`learning_rate`**: –°–∫–æ—Ä–æ—Å—Ç—å –æ–±—É—á–µ–Ω–∏—è. –û–ø—Ä–µ–¥–µ–ª—è–µ—Ç, –Ω–∞—Å–∫–æ–ª—å–∫–æ —Å–∏–ª—å–Ω–æ –æ–±–Ω–æ–≤–ª—è—é—Ç—Å—è –≤–µ—Å–∞ –Ω–∞ –∫–∞–∂–¥–æ–º —à–∞–≥–µ.
- **`num_iterations`**: –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –∏—Ç–µ—Ä–∞—Ü–∏–π –æ–±—É—á–µ–Ω–∏—è.
- **`weights`**: –í–µ—Å–∞ –º–æ–¥–µ–ª–∏ (–∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –ø—Ä–∏–∑–Ω–∞–∫–∞). –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É—é—Ç—Å—è –∫–∞–∫ `None`.
- **`bias`**: –°–≤–æ–±–æ–¥–Ω—ã–π —á–ª–µ–Ω (—Å–º–µ—â–µ–Ω–∏–µ). –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ—Ç—Å—è –∫–∞–∫ `None`.

---

#### 2. –°–∏–≥–º–æ–∏–¥–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è (`sigmoid`)
```python
def sigmoid(self, z):
    return 1 / (1 + np.exp(-z))
```
- **–°–∏–≥–º–æ–∏–¥–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è** –ø—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç –ª—é–±–æ–µ —á–∏—Å–ª–æ –≤ –∏–Ω—Ç–µ—Ä–≤–∞–ª $\([0, 1]\)$. –≠—Ç–æ –ø–æ–∑–≤–æ–ª—è–µ—Ç –∏–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∏—Ä–æ–≤–∞—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç –∫–∞–∫ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å.
- –§–æ—Ä–º—É–ª–∞:
  $\[
  \sigma(z) = \frac{1}{1 + e^{-z}}
  \]$

---

#### 3. –§—É–Ω–∫—Ü–∏—è –ø–æ—Ç–µ—Ä—å (`log_loss`)
```python
def log_loss(self, y, y_pred):
    return -np.mean(y * np.log(y_pred) + (1 - y) * np.log(1 - y_pred))
```
- **–§—É–Ω–∫—Ü–∏—è –ø–æ—Ç–µ—Ä—å (Log Loss)** –∏–∑–º–µ—Ä—è–µ—Ç, –Ω–∞—Å–∫–æ–ª—å–∫–æ –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–Ω—ã–µ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ –æ—Ç–ª–∏—á–∞—é—Ç—Å—è –æ—Ç —Ä–µ–∞–ª—å–Ω—ã—Ö –º–µ—Ç–æ–∫.
- –§–æ—Ä–º—É–ª–∞:
  $\[
  \text{Log Loss} = -\frac{1}{N} \sum_{i=1}^{N} \left[ y_i \log(\hat{y}_i) + (1 - y_i) \log(1 - \hat{y}_i) \right]
  \]$
  –ì–¥–µ:
  - $\( y_i \)$ ‚Äî —Ä–µ–∞–ª—å–Ω–∞—è –º–µ—Ç–∫–∞ (0 –∏–ª–∏ 1),
  - $\( \hat{y}_i \)$ ‚Äî –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–Ω–∞—è –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å.

---

#### 4. –û–±—É—á–µ–Ω–∏–µ –º–æ–¥–µ–ª–∏ (`fit`)
```python
def fit(self, X, y):
    n_samples, n_features = X.shape
    self.weights = np.zeros(n_features)
    self.bias = 0
```
- **–ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –≤–µ—Å–æ–≤ –∏ —Å–º–µ—â–µ–Ω–∏—è**:
  - –í–µ—Å–∞ (`weights`) –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É—é—Ç—Å—è –Ω—É–ª—è–º–∏.
  - –°–º–µ—â–µ–Ω–∏–µ (`bias`) –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ—Ç—Å—è –Ω—É–ª—ë–º.

---

#### –ì—Ä–∞–¥–∏–µ–Ω—Ç–Ω—ã–π —Å–ø—É—Å–∫
```python
for _ in range(self.num_iterations):
    linear_model = np.dot(X, self.weights) + self.bias
    y_pred = self.sigmoid(linear_model)
```
- **–õ–∏–Ω–µ–π–Ω–∞—è –∫–æ–º–±–∏–Ω–∞—Ü–∏—è**:
  - –í—ã—á–∏—Å–ª—è–µ—Ç—Å—è –ª–∏–Ω–µ–π–Ω–∞—è –∫–æ–º–±–∏–Ω–∞—Ü–∏—è –ø—Ä–∏–∑–Ω–∞–∫–æ–≤ –∏ –≤–µ—Å–æ–≤:
    $\[
    z = X \cdot \text{weights} + \text{bias}
    \]$
- **–ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–µ**:
  - –õ–∏–Ω–µ–π–Ω–∞—è –∫–æ–º–±–∏–Ω–∞—Ü–∏—è –ø—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç—Å—è –≤ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å —Å –ø–æ–º–æ—â—å—é —Å–∏–≥–º–æ–∏–¥–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏:
    $\[
    \hat{y} = \sigma(z)
    \]$

---

#### –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –≤–µ—Å–æ–≤ –∏ —Å–º–µ—â–µ–Ω–∏—è

- **–ì—Ä–∞–¥–∏–µ–Ω—Ç—ã**:
  - `dw`: –ì—Ä–∞–¥–∏–µ–Ω—Ç –ø–æ –≤–µ—Å–∞–º. –ü–æ–∫–∞–∑—ã–≤–∞–µ—Ç, –∫–∞–∫ –Ω—É–∂–Ω–æ –∏–∑–º–µ–Ω–∏—Ç—å –≤–µ—Å–∞, —á—Ç–æ–±—ã —É–º–µ–Ω—å—à–∏—Ç—å –æ—à–∏–±–∫—É.
  - `db`: –ì—Ä–∞–¥–∏–µ–Ω—Ç –ø–æ —Å–º–µ—â–µ–Ω–∏—é. –ü–æ–∫–∞–∑—ã–≤–∞–µ—Ç, –∫–∞–∫ –Ω—É–∂–Ω–æ –∏–∑–º–µ–Ω–∏—Ç—å —Å–º–µ—â–µ–Ω–∏–µ, —á—Ç–æ–±—ã —É–º–µ–Ω—å—à–∏—Ç—å –æ—à–∏–±–∫—É.
- **–û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –≤–µ—Å–æ–≤ –∏ —Å–º–µ—â–µ–Ω–∏—è**:
  - –í–µ—Å–∞ –∏ —Å–º–µ—â–µ–Ω–∏–µ –æ–±–Ω–æ–≤–ª—è—é—Ç—Å—è —Å —É—á—ë—Ç–æ–º –≥—Ä–∞–¥–∏–µ–Ω—Ç–æ–≤ –∏ —Å–∫–æ—Ä–æ—Å—Ç–∏ –æ–±—É—á–µ–Ω–∏—è:
    $weights = weights - learning_rate \cdot dw$
    $bias = bias - learning_rate \cdot db$
